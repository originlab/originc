Description

Get a node's tagname.

Syntax

string tree_get_enum_node_name( TreeNode & trNode, string strPrefix )

Parameters

trNode
[input] the tree node to get enumerating node name
strPrefix
[input] prefix string

Return

string of enumerating node name

Examples

EX1

#include <GetNBox.h>

void tree_get_enum_node_name_ex1(TreeNode& tr)
{        
    tr.AddNode("Node1");
    tr.AddNode("Node2");
    tr.AddNode("Node3");
 
    string strNextTagName;
    strNextTagName = tree_get_enum_node_name(tr, "Node");
    
    out_str(strNextTagName); //-->strNextTagName="Node4"
}
