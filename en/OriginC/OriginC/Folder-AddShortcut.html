Description

Add a shortcut to the Folder

Syntax

BOOL AddShortcut( PageBase & page )

Parameters

page
[input] the page to which shortcut needs to be created

Return

Returns FALSE if already present or invalid page. Otherwise TRUE

Examples

EX1 

void    AddShortcut_Ex1()
{
    Folder fld = Project.ActiveFolder();
    
    if(fld.IsValid())
    {
        PageBase pb;
        pb = Project.Pages();
        BOOL bSuccess=false;
        if(pb.IsValid())
            bSuccess=fld.AddShortcut(pb);
        printf("Is Page has add shortcut:%s",bSuccess?"yes":"not");
        return;
    }
    printf("there is no active folder in current project");
}
