Description

Get list of cells with Notes

Syntax

int GetCellsWithNotes( vector<int> & vCols, vector<int> & vRows )

Parameters

vCols
[output] column number(0-offset) of cells with notes
vRows
[output] row number(0-offset) of cells with notes

Return

number of cells with notes

Examples

EX1

void GetCellsWithNotes_ex()
{
        Worksheet wks = Project.ActiveLayer();
        vector<int> vcols, vrows;
        int nn = wks.GetCellsWithNotes(vcols, vrows);
        for(int ii = 0; ii < nn; ii++) {
                string strNote;
                wks.GetCellComment(vrows[ii], vcols[ii], strNote);
                //strNote might have CRLF, and need to end with \r\n here
                printf("%d. row,col=%d,%d:(%s)\r\n", ii+1, vrows[ii]+1, vcols[ii]+1, strNote);
        }
}
